// --- เอาลิงก์ Web App ที่ Deploy ล่าสุดมาใส่ตรงนี้ --- const API_URL = "https://script.google.com/macros/s/AKfycbwH6kT_7xgH0YGytHLENKBmNY5yD34syaMwzFu_nlhbbBlxjZuic-KgOB4S37JSepE7/exec"; function changeStep(n) { document.querySelectorAll('.step').forEach(el => el.classList.remove('active')); document.getElementById('step' + n).classList.add('active'); } // สร้างฟังก์ชันกลางสำหรับยิงข้อมูลไปหา Google Apps Script function sendDataToGoogle(actionName, payloadData, loadingElementId, onSuccess) { document.getElementById(loadingElementId).style.display = 'block'; fetch(API_URL, { method: 'POST', // ใช้ text/plain เพื่อป้องกันปัญหา CORS Policy ข้ามโดเมน headers: { 'Content-Type': 'text/plain;charset=utf-8' }, body: JSON.stringify({ action: actionName, payload: payloadData }) }) .then(response => response.json()) .then(result => { document.getElementById(loadingElementId).style.display = 'none'; if(result.status === "success") { onSuccess(result); } else { alert("เกิดข้อผิดพลาด: " + result.message); } }) .catch(error => { document.getElementById(loadingElementId).style.display = 'none'; alert("ไม่สามารถเชื่อมต่อเซิร์ฟเวอร์ได้"); console.error(error); }); } function submitStep1() { var data = { subjectId: document.getElementById('subjectId').value, subjectName: document.getElementById('subjectName').value }; if(!data.subjectId || !data.subjectName) { alert("กรุณากรอกให้ครบค่ะ"); return; } sendDataToGoogle("saveSubject", data, 'loading1', function() { document.getElementById('showSubjectId').innerText = data.subjectId; changeStep(2); }); } function submitStep2() { var data = { subjectId: document.getElementById('subjectId').value, groupId: document.getElementById('groupId').value, groupName: document.getElementById('groupName').value }; if(!data.groupId || !data.groupName) { alert("กรุณากรอกให้ครบค่ะ"); return; } sendDataToGoogle("saveGroup", data, 'loading2', function() { document.getElementById('showGroupId').innerText = data.groupId; changeStep(3); }); } function submitStep3() { var rawData = document.getElementById('studentList').value; if(!rawData.trim()) { alert("กรุณาวางรายชื่อนิสิตก่อนค่ะ"); return; } var lines = rawData.split('\n'); var studentArray = []; for(var i = 0; i < lines.length; i++) { var line = lines[i].trim(); if(line) { var parts = line.split(/\s+/); var studentId = parts[0]; var studentName = parts.slice(1).join(" "); studentArray.push([studentId, studentName]); } } var data = { groupId: document.getElementById('groupId').value, students: studentArray }; sendDataToGoogle("saveStudents", data, 'loading3', function() { changeStep(4); }); }
